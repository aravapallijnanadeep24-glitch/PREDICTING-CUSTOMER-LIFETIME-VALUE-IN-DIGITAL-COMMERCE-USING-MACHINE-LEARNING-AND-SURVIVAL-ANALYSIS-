Feature Engineering¶

In [26]:
snapshot_date = historical['InvoiceDate'].max() + pd.Timedelta(days=1)
In [27]:
rfm = historical.groupby('CustomerID').agg({
    'InvoiceDate': lambda x:
        (snapshot_date - x.max()).days,
    'Invoice': 'nunique',
    'TotalPrice': 'sum'
}).reset_index()
In [28]:
rfm.columns = [
    'CustomerID',
    'Recency',
    'Frequency',
    'Monetary'
]

Tenure¶

In [29]:
tenure = historical.groupby(
    'CustomerID'
)['InvoiceDate'].agg(
    ['min', 'max']
).reset_index()
In [30]:
tenure['Tenure'] = (
    tenure['max'] - tenure['min']
).dt.days
In [31]:
rfm = rfm.merge(
    tenure[['CustomerID', 'Tenure']],
    on='CustomerID'
)
In [32]:
# Average Order Value
aov = historical.groupby(
    'CustomerID'
).agg({
    'TotalPrice': 'sum',
    'Invoice': 'nunique'
}).reset_index()

aov['AvgOrderValue'] = (
    aov['TotalPrice']
    /
    aov['Invoice']
)

rfm = rfm.merge(
    aov[['CustomerID', 'AvgOrderValue']],
    on='CustomerID'
)
In [33]:
# Total Quantity Purchased

qty = historical.groupby(
    'CustomerID'
)['Quantity'].sum().reset_index()

qty.columns = [
    'CustomerID',
    'TotalQuantity'
]

rfm = rfm.merge(
    qty,
    on='CustomerID'
)
In [34]:
temp = historical.sort_values(
    ['CustomerID', 'InvoiceDate']
)

temp['PreviousDate'] = temp.groupby(
    'CustomerID'
)['InvoiceDate'].shift(1)

temp['Gap'] = (
    temp['InvoiceDate']
    -
    temp['PreviousDate']
).dt.days
In [35]:
# Standard Deviation of Purchase Gap

std_gap = temp.groupby(
    'CustomerID'
)['Gap'].std().reset_index()

std_gap.columns = [
    'CustomerID',
    'StdGap'
]

rfm = rfm.merge(
    std_gap,
    on='CustomerID',
    how='left'
)

rfm['StdGap'] = rfm['StdGap'].fillna(0)
In [36]:
# Purchase Momentum
momentum = historical.groupby(
    'CustomerID'
).agg({
    'Invoice':'nunique'
}).reset_index()

momentum['Momentum'] = (
    momentum['Invoice']
    /
    (rfm['Tenure'] + 1)
)

momentum = momentum[
    ['CustomerID','Momentum']
]

rfm = rfm.merge(
    momentum,
    on='CustomerID'
)
